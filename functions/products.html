商品一覧情報の取得

商品一覧の取得

Query

query {
  products(limit: 5, sortkey: "create_date", sorttype: "d") {
    nodes {
      id
      name
      create_date
      ProductImage {
        file_name
      }
      ProductClasses {
        code
        stock
        price01
      }
      ProductCategories {
        Category {
          id
          name
        }
      }
    }
    totalCount
    pageInfo {
      hasNextPage
      hasPreviousPage
    }
  }
}
